Очередь - контейнер адаптор (т.е. выбирается из другого контейнера)
```cpp
#include <deque>
#include <iostream>
#include <queue>

using namespace std;

int main() {
  deque<int> d{10, 20, 30, 40, 50, 60};
  queue<int> q{d};

  q.emplace(70);

  cout << d[1] << endl;
  // q[1]; // не сработает
  
  while (!q.empty()) {
    cout << q.front() << "\t size -> " << q.size() << '\n';
    q.pop();
  }
  return 0;
}
```
